```bash
#!/usr/bin/env bash
set -euo pipefail

# arch/quickshell.sh — Install Quickshell + ddcutil + i2c-tools, configure i2c, symlink config.
# Pattern: mirrors arch/waybar.sh (REPO_ROOT, PACKAGES array, main dispatcher, [LABEL] echos).
# Divergence from waybar.sh: uses a single directory symlink instead of per-file copies (D-17).

REPO_ROOT="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
QS_SRC="$REPO_ROOT/.config/quickshell"
QS_DST="$HOME/.config/quickshell"

PACKAGES=(
  quickshell
  ddcutil
  i2c-tools
)

install_packages() {
  echo "[INSTALL] quickshell and runtime dependencies (${PACKAGES[*]})"
  sudo pacman -Sy --noconfirm --needed "${PACKAGES[@]}"
}

setup_i2c() {
  echo "[CONFIG] i2c kernel module and group membership"
  sudo modprobe i2c-dev
  sudo usermod -aG i2c "$USER"
  echo "i2c-dev" | sudo tee /etc/modules-load.d/i2c.conf > /dev/null
  echo "[CONFIG] /etc/modules-load.d/i2c.conf written (i2c-dev will load on every boot)"
}

symlink_config() {
  echo "[CONFIG] symlink $QS_DST -> $QS_SRC"
  mkdir -p "$(dirname "$QS_DST")"
  rm -rf "$QS_DST"
  ln -s "$QS_SRC" "$QS_DST"
}

verify_install() {
  echo "[VERIFY] checking install state"
  command -v quickshell >/dev/null || { echo "[VERIFY] FAIL: quickshell not in PATH"; exit 1; }
  command -v ddcutil    >/dev/null || { echo "[VERIFY] FAIL: ddcutil not in PATH";    exit 1; }
  test -L "$QS_DST"                 || { echo "[VERIFY] FAIL: $QS_DST is not a symlink"; exit 1; }
  test -f "$QS_DST/shell.qml"       || { echo "[VERIFY] FAIL: $QS_DST/shell.qml not reachable through symlink"; exit 1; }
  test -f /etc/modules-load.d/i2c.conf || { echo "[VERIFY] FAIL: /etc/modules-load.d/i2c.conf missing"; exit 1; }
  echo "[VERIFY] OK"
}

print_summary() {
  echo "[DONE] Quickshell installed and configured."
  echo "[DONE] Run \`quickshell\` from a terminal to launch the bar (Hyprland exec-once is intentionally not modified in Phase 12)."
  echo ""
  echo "Log out and back in for i2c group to take effect (required for ddcutil)"
}

main() {
  install_packages
  setup_i2c
  symlink_config
  verify_install
  print_summary
}

main "$@"
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User shell → sudo (root) | Script invokes `sudo pacman`, `sudo modprobe`, `sudo usermod`, `sudo tee`. All operations are well-defined Arch system management commands |
| Repo filesystem → `~/.config/quickshell` | Script creates a symlink from `$HOME` into the repo. Repo is user-owned and version-controlled — no untrusted writes cross this boundary |
| Pacman repository → local system | Standard pacman install — Arch's [extra] repository, signed packages |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-06 | Elevation of Privilege | sudo pacman / modprobe / usermod / tee | accept | Standard Arch install pattern, identical to `arch/waybar.sh`. Each sudo invocation is scoped to a single command. Pacman packages are signed and trusted; `i2c-dev` is a standard kernel module; group membership change is minimal privilege (i2c bus access only). |
| T-12-07 | Tampering | `rm -rf "$QS_DST"` (symlink overwrite) | mitigate | `QS_DST` is constructed as `$HOME/.config/quickshell` — fixed path, not user-input. `set -u` prevents `$HOME` being empty. The `rm -rf` only removes a single directory or symlink at a known path; it cannot escape `$HOME/.config/`. Verifiable via `grep` that `QS_DST` is the only argument to `rm -rf`. |
| T-12-08 | Denial of Service | i2c-dev modprobe failure on non-Arch / non-i2c-capable system | accept | Script targets Arch on x86_64 desktop with DDC/CI-capable monitor (project's documented target hardware). On unsupported hardware, `modprobe i2c-dev` fails fast under `set -e` and the user sees a clear error. |
| T-12-09 | Information Disclosure | Script writes to /etc/modules-load.d/ | accept | Content is the literal string `i2c-dev`. No secrets, no user input, no PII. Standard system configuration. |
| T-12-10 | Tampering | Hyprland exec-once / Waybar config modification | mitigate | Script explicitly does not touch `.config/hypr/` or `.config/waybar/` (D-19, D-20, BAR-06). Verified by `! grep -q "exec-once\\|hyprland\\|waybar" arch/quickshell.sh` in acceptance_criteria. |
